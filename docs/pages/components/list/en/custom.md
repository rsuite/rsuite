class ListDemo extends React.Component {
  constructor() {
    super();
    this.renderRaise = this.renderRaise.bind(this);
  }

  renderRaise(number) {
    const isPositive = number > 0;
    const isNegative = number < 0;
    return (
      <span style={{
        paddingLeft: 15,
        color: isNegative ? 'red' : 'green'
